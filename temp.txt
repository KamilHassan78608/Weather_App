    <script>
        // --- CONFIGURATION ---
        // !! IMPORTANT: Replace with your own OpenWeatherMap API key !!
        const API_KEY = "9923acc08c7c027d6e965eb28da9f12d"; 
        // ---------------------

        // Get references to all the DOM elements
        const cityInput = document.getElementById('city-input');
        const searchButton = document.getElementById('search-button');
        const messageBox = document.getElementById('message-box');
        const welcomeMessage = document.getElementById('welcome-message');
        const weatherDisplay = document.getElementById('weather-display');

        // Main data fields
        const cityName = document.getElementById('city-name');
        const weatherIcon = document.getElementById('weather-icon');
        const temperature = document.getElementById('temperature');
        const description = document.getElementById('description');
        const feelsLike = document.getElementById('feels-like');

        // Detail grid fields
        const humidity = document.getElementById('humidity');
        const windSpeed = document.getElementById('wind-speed');
        const pressure = document.getElementById('pressure');
        const visibility = document.getElementById('visibility');
        const tempMin = document.getElementById('temp-min');
        const tempMax = document.getElementById('temp-max');
        const sunrise = document.getElementById('sunrise');
        const sunset = document.getElementById('sunset');

        // Add event listener for the search button
        searchButton.addEventListener('click', fetchWeather);
        // Add event listener for pressing 'Enter' in the input
        cityInput.addEventListener('keyup', (event) => {
            if (event.key === 'Enter') {
                fetchWeather();
            }
        });

        // Function to show messages (error, loading, success)
        function showMessage(message, type = 'loading') {
            messageBox.textContent = message;
            messageBox.classList.remove('hidden', 'bg-red-500', 'bg-blue-500', 'bg-yellow-500');

            if (type === 'error') {
                messageBox.classList.add('bg-red-500');
            } else if (type === 'loading') {
                messageBox.classList.add('bg-blue-500');
            } else if (type === 'warn') {
                 messageBox.classList.add('bg-yellow-500', 'text-black');
            }
            messageBox.classList.remove('hidden');
        }

        // Function to hide the message box
        function hideMessage() {
            messageBox.classList.add('hidden');
        }

        // Main function to fetch weather data
        async function fetchWeather() {
            const city = cityInput.value;

            // --- Validation ---
            if (API_KEY === "YOUR_API_KEY_HERE" || !API_KEY) {
                showMessage("Please add your OpenWeatherMap API key to the script.", 'warn');
                return;
            }
            if (!city) {
                showMessage("Please enter a city name.", 'error');
                return;
            }

            // --- Start Loading ---
            showMessage("Fetching weather data...", 'loading');
            welcomeMessage.classList.add('hidden');
            weatherDisplay.classList.add('hidden');

            const apiUrl = `https://api.openweathermap.org/data/2.5/weather?q=${city}&appid=${API_KEY}&units=metric`;

            try {
                const response = await fetch(apiUrl);
                
                if (!response.ok) {
                    if (response.status === 404) {
                        throw new Error(`City not found: "${city}"`);
                    } else if (response.status === 401) {
                        throw new Error("Invalid API Key. Please check your key in the script.");
                    } else {
                        throw new Error(`Error: ${response.statusText}`);
                    }
                }

                const data = await response.json();
                
                // --- Success ---
                hideMessage();
                displayWeather(data);

            } catch (error) {
                console.error("Fetch Error:", error);
                showMessage(error.message, 'error');
            }
        }

        // Function to update the DOM with new weather data
        function displayWeather(data) {
            // Main info
            cityName.textContent = `${data.name}, ${data.sys.country}`;
            temperature.textContent = `${Math.round(data.main.temp)}°C`;
            description.textContent = data.weather[0].description;
            feelsLike.textContent = `Feels like ${Math.round(data.main.feels_like)}°C`;
            weatherIcon.src = `https://openweathermap.org/img/wn/${data.weather[0].icon}@4x.png`;
            weatherIcon.alt = data.weather[0].description;

            // Detail grid
            humidity.textContent = `${data.main.humidity}%`;
            windSpeed.textContent = `${data.wind.speed.toFixed(1)} m/s`;
            pressure.textContent = `${data.main.pressure} hPa`;
            visibility.textContent = `${(data.visibility / 1000).toFixed(1)} km`;
            tempMin.textContent = `${Math.round(data.main.temp_min)}°C`;
            tempMax.textContent = `${Math.round(data.main.temp_max)}°C`;
            
            // Format sunrise/sunset times
            sunrise.textContent = formatTime(data.sys.sunrise, data.timezone);
            sunset.textContent = formatTime(data.sys.sunset, data.timezone);

            // Show the weather display
            weatherDisplay.classList.remove('hidden');
        }

        // Helper function to format UNIX timestamps to local time
        function formatTime(unixTimestamp, timezoneOffset) {
            // Convert to milliseconds and adjust for timezone offset
            const date = new Date((unixTimestamp + timezoneOffset) * 1000);
            // Get UTC hours and minutes
            const hours = date.getUTCHours().toString().padStart(2, '0');
            const minutes = date.getUTCMinutes().toString().padStart(2, '0');
            return `${hours}:${minutes}`;
        }

    </script>
